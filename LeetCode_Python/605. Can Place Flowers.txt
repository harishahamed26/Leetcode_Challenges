class Solution(object):
    def canPlaceFlowers(self, flowerbed, n):
        """
        :type flowerbed: List[int]
        :type n: int
        :rtype: bool
        """
        if n == 0:
            return True
        length = len(flowerbed)
        for i in range(length):
            if flowerbed[i] == 0:
                if i == 0 or flowerbed[i-1] == 0:
                    if i == length -1 or flowerbed[i+1] == 0:
                        flowerbed[i] = 1
                        n -= 1
                if n == 0:
                    return True
                
        if n != 0:
            return False
